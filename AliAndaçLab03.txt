import java.util.Scanner;

class Insan{
        public String ad,soyad,sacRengi,gozRengi;
        public int yas;
        public double boy,kilo;
        

        public Insan(){
            
        }

        public Insan(String ad,String soyad){
        this.ad = ad;
        this.soyad = soyad;

        }


        public Insan(String ad, String soyad, String sacRengi, String gozRengi, int yas, double kilo, double boy) {
        this.ad = ad;
        this.soyad = soyad;
        this.sacRengi = sacRengi;
        this.gozRengi = gozRengi;
        this.yas = yas;
        this.kilo = kilo;
        this.boy = boy;
    }

    public void adYaz() {
        System.out.println("Ad Soyad: " + ad);
    }
     public void soyadYaz() {
        System.out.println("Soyad: " + soyad);
    }
    public void gozRengiYaz() {
        System.out.println("Göz Rengi: " + gozRengi);
    }

    public void sacRengiYaz() {
        System.out.println("Saç Rengi: " + sacRengi);
    }

    public void yasYaz() {
        System.out.println("Yaş: " + yas);
    }

    public void kiloYaz() {
        System.out.println("Kilo: " + kilo + " kg");
    }

    public void boyYaz() {
        System.out.println("Boy: " + boy + " m");
    }

}
